### solution
```java
import java.util.*;

class Permutation {
    static int cnt = 0;
    public static void perm(int[] arr, int depth, int n, int r, int[][] dungeons, int k){
        if (depth == r) { // 한 사이클 끝나면 만들어진 arr로 확인하기
            System.out.println(Arrays.toString(arr));
            return;
        }
        
        for(int i = depth; i < n; i++){
            swap(arr, i, depth); 
            perm(arr, depth + 1, n, r, dungeons, k);
            swap(arr, i, depth);
            
        }

    }
    
    private static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
    
    
    public static int solution(int k, int[][] dungeons) {
        int answer = -1;
        int arr[] = new int[dungeons.length];
        for(int i = 0; i < dungeons.length; i++){
            arr[i] = i;
        }
        perm(arr, 0, arr.length, arr.length, dungeons, k);
        return answer;
    }

    public static void main(String[] args){
      int[][] dungeons = {{80, 20}, {50, 40}, {30, 10}};
      solution(80, dungeons);
    }



    public static boolean dungeon(int[] arr, int[][] dungeons, int k){
      System.out.println(Arrays.toString(arr));
      for(int i = 0; i < arr.length; i++){
        int first = dungeons[arr[i]][0];
        int second = dungeons[arr[i]][1];

        if( k <= first ){
          k -= second;
        }
        else {
          break;
        }
        System.out.println("first" + first + "second" + second);
        System.out.println("k" + k);
      }
      
      if (k > 0) return true;
      else return false;
    }
}
```
