# 프로그래머스 완주하지 못한 선수
https://school.programmers.co.kr/learn/courses/30/lessons/42576

### solution
```java
import java.lang.*;
import java.util.*;

class Solution {
    public String solution(String[] participant, String[] completion) {
        String answer = "";
        Map<String, Integer> map = new HashMap<>();
        
        for(int i = 0; i < participant.length; i++){ //초기화
            String str = participant[i];
            if (map.get(str) == null) map.put(str, 1); // 입력되지 않았으면 put하고 1로 초기화
            else map.replace(str, map.get(str) + 1); // 값 있으면 원래 값에서 +1
        }
        
        for(int i = 0; i < completion.length; i++){
            if (map.containsKey(completion[i])){ //map에 포함된 key값이 있으면
                map.replace(completion[i], map.get(completion[i])-1); //1씩 빼기
            }
        }
        
        for(Map.Entry<String, Integer> entrySet : map.entrySet()) {
            if (entrySet.getValue() > 0){ //0보다 큰 수 있으면 
                answer += entrySet.getKey(); //answer 에 더하기
            }
        }
        return answer;
    }
}
```
