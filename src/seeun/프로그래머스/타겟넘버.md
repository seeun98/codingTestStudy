### solution
- 재귀
```java
class Solution {
    int target;
    int[] numbers;
    int answer = 0;
    public int solution(int[] numbers, int target) {
        
        this.target = target;
        this.numbers = numbers;
        DFS(0, 0);
        return answer;
    }
    
    public void DFS(int index, int sum) {
        if (index == numbers.length) {
            if(sum == target) answer++;
            return;
        }

        DFS(index+1, sum + numbers[index]);
        DFS(index+1, sum - numbers[index]);
                
    }
}
```
